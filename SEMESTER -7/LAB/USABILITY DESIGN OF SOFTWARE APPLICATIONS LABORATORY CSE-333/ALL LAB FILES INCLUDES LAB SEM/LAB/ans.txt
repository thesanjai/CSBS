import androidx.appcompat.app.AppCompatActivity;
import android.graphics.Color;
import android.os.Bundle;
import android.view.View;
import android.widget.RelativeLayout;
import android.widget.Button;

public class MainActivity extends AppCompatActivity {
    RelativeLayout root;
    Button btnToggle;
    boolean state = false;

    @Override
    protected void onCreate(Bundle s) {
        super.onCreate(s);
        setContentView(R.layout.activity_main);
        root = findViewById(R.id.root);
        btnToggle = findViewById(R.id.btnToggle);

        btnToggle.setOnClickListener(v -> {
            if (!state) {
                // show image background (ensure drawable exists)
                root.setBackgroundResource(R.drawable.bg_image);
            } else {
                // show plain color
                root.setBackgroundColor(Color.parseColor("#FFEB3B")); // yellow
            }
            state = !state;
        });
    }
}

package com.example.entercounter;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.KeyEvent;
import android.widget.EditText;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    EditText etInput;
    TextView tvCount;
    int enterCount = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etInput = findViewById(R.id.etInput);
        tvCount = findViewById(R.id.tvCount);

        etInput.setOnKeyListener((v, keyCode, event) -> {
            if (event.getAction() == KeyEvent.ACTION_DOWN && keyCode == KeyEvent.KEYCODE_ENTER) {
                enterCount++;
                tvCount.setText("Enter Key Pressed: " + enterCount + " times");
                return true; // consume event
            }
            return false;
        });
    }
}

package com.example.canteen;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.Menu;
import android.view.MenuItem;
import android.widget.ArrayAdapter;
import android.widget.ListView;
import java.util.ArrayList;

public class MainActivity extends AppCompatActivity {

    ListView listMenu;
    ArrayList<String> items;
    ArrayAdapter<String> adapter;

    @Override
    protected void onCreate(Bundle s) {
        super.onCreate(s);
        setContentView(R.layout.activity_main);

        listMenu = findViewById(R.id.listMenu);

        items = new ArrayList<>();
        // sample items with category tags in name
        items.add("Tea - ₹10 (snack)");
        items.add("Samosa - ₹20 (snack)");
        items.add("Idli - ₹30 (meal)");
        items.add("Veg Biryani - ₹80 (meal)");
        items.add("Coffee - ₹15 (snack)");

        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, items);
        listMenu.setAdapter(adapter);
    }

    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        getMenuInflater().inflate(R.menu.menu_main, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(MenuItem item) {
        String id = item.getTitle().toString();
        ArrayList<String> filtered = new ArrayList<>();
        if (id.equals("All")) {
            adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, items);
        } else if (id.equals("Snacks")) {
            for (String s : items) if (s.contains("(snack)")) filtered.add(s);
            adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, filtered);
        } else if (id.equals("Meals")) {
            for (String s : items) if (s.contains("(meal)")) filtered.add(s);
            adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, filtered);
        }
        listMenu.setAdapter(adapter);
        return super.onOptionsItemSelected(item);
    }
}

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.content.Intent;

import android.widget.Button;
import android.widget.EditText;

public class MainActivity extends AppCompatActivity {
    EditText etPhone, etMsg;
    Button btnSend;

    @Override
    protected void onCreate(Bundle s) {
        super.onCreate(s);
        setContentView(R.layout.activity_main);
        etPhone = findViewById(R.id.etPhone);
        etMsg = findViewById(R.id.etMsg);
        btnSend = findViewById(R.id.btnSend);

        btnSend.setOnClickListener(v -> {

            String msg = etMsg.getText().toString();

            intent.putExtra("sms_body", msg);
            startActivity(intent);
        });
    }
}

import androidx.appcompat.app.AppCompatActivity;
import androidx.fragment.app.Fragment;
import android.os.Bundle;
import android.widget.Button;

public class MainActivity extends AppCompatActivity {

    Button btnOne, btnTwo;

    @Override
    protected void onCreate(Bundle s) {
        super.onCreate(s);
        setContentView(R.layout.activity_main);

        btnOne = findViewById(R.id.btnOne);
        btnTwo = findViewById(R.id.btnTwo);

        // load default fragment
        getSupportFragmentManager().beginTransaction()
                .replace(R.id.fragment_container, new FragmentOne())
                .commit();

        btnOne.setOnClickListener(v -> replace(new FragmentOne()));
        btnTwo.setOnClickListener(v -> replace(new FragmentTwo()));
    }

    void replace(Fragment f) {
        getSupportFragmentManager().beginTransaction()
                .replace(R.id.fragment_container, f)
                .commit();
    }
}

package com.example.loginsqlite;

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.*;

public class DatabaseHelper extends SQLiteOpenHelper {

    private static final String DB_NAME = "LoginDB.db";
    private static final int DB_VER = 1;
    private static final String TABLE_USERS = "users";

    public DatabaseHelper(Context c) {
        super(c, DB_NAME, null, DB_VER);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        db.execSQL("CREATE TABLE " + TABLE_USERS + " (id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT UNIQUE, password TEXT)");
        // demo user
        ContentValues cv = new ContentValues();
        cv.put("username", "admin");
        cv.put("password", "1234");
        db.insert(TABLE_USERS, null, cv);
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldV, int newV) {
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_USERS);
        onCreate(db);
    }

    public boolean insertUser(String username, String password) {
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues cv = new ContentValues();
        cv.put("username", username);
        cv.put("password", password);
        long r = db.insert(TABLE_USERS, null, cv);
        return r != -1;
    }

    public boolean checkUser(String username, String password) {
        SQLiteDatabase db = this.getReadableDatabase();
        Cursor cursor = db.rawQuery("SELECT * FROM " + TABLE_USERS + " WHERE username=? AND password=?", new String[]{username, password});
        boolean exists = (cursor.getCount() > 0);
        cursor.close();
        return exists;
    }
}

                tvStatus.setText("Enter credentials");
                return;
            }
            if (db.checkUser(u, p)) {
                tvStatus.setText("Login Successful");
            } else {
                tvStatus.setText("Invalid credentials");
            }
        });

        btnRegister.setOnClickListener(v -> startActivity(new Intent(MainActivity.this, RegisterActivity.class)));
    }
}


java/com/example/loginsqlite/RegisterActivity.java

package com.example.loginsqlite;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.*;

public class RegisterActivity extends AppCompatActivity {

    EditText etNewUser, etNewPass;
    Button btnCreate;
    TextView tvRegStatus;
    DatabaseHelper db;

    @Override
    protected void onCreate(Bundle s) {
        super.onCreate(s);
        setContentView(R.layout.activity_register);

        etNewUser = findViewById(R.id.etNewUser);
        etNewPass = findViewById(R.id.etNewPass);
        btnCreate = findViewById(R.id.btnCreate);
        tvRegStatus = findViewById(R.id.tvRegStatus);

        db = new DatabaseHelper(this);

        btnCreate.setOnClickListener(v -> {
            String u = etNewUser.getText().toString().trim();
            String p = etNewPass.getText().toString().trim();
            if (u.isEmpty() || p.isEmpty()) {
                tvRegStatus.setText("Enter details");
                return;
            }
            boolean inserted = db.insertUser(u, p);
            tvRegStatus.setText(inserted ? "Registered successfully" : "Error: username may already exist");
        });
    }
}

<activity android:name=".RegisterActivity" />

Quick tips for all apps

Replace package com.example.xxx; with your actual package if different.

Put image files into app/src/main/res/drawable/.

For activities other than MainActivity, declare them inside AndroidManifest.xml.

Clean & Rebuild project if you get resource errors.

If you use runtime permissions (e.g., actual SMS send), request them properly — sample apps above use Intent which avoids SMS permission.
