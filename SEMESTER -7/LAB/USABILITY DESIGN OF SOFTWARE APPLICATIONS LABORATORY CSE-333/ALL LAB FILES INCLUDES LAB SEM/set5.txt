package com.example.colorchanger;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.graphics.Color;
import android.widget.Button;
import android.widget.LinearLayout;
import android.widget.TextView;
import java.util.Random;

public class MainActivity extends AppCompatActivity {

    LinearLayout mainLayout;
    TextView tvMessage;
    Button btnChange;
    Random random = new Random();

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        mainLayout = findViewById(R.id.mainLayout);
        tvMessage = findViewById(R.id.tvMessage);
        btnChange = findViewById(R.id.btnChange);

        btnChange.setOnClickListener(v -> {
            // Generate random RGB values
            int red = random.nextInt(256);
            int green = random.nextInt(256);
            int blue = random.nextInt(256);

            // Apply new background color
            int bgColor = Color.rgb(red, green, blue);
            mainLayout.setBackgroundColor(bgColor);

            // Generate another random color for text
            int textRed = random.nextInt(256);
            int textGreen = random.nextInt(256);
            int textBlue = random.nextInt(256);
            int textColor = Color.rgb(textRed, textGreen, textBlue);

            tvMessage.setTextColor(textColor);
            tvMessage.setText("Colors Changed!");
        });
    }
}
