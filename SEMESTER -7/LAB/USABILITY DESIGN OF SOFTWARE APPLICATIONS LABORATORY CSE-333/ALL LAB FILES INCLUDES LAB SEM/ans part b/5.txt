package com.kasinadhuni.set5parta;

import androidx.appcompat.app.AppCompatActivity;
import android.graphics.Color;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.LinearLayout;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    LinearLayout mainLayout;
    TextView tvMessage;
    Button btnChangeColor;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        mainLayout = findViewById(R.id.mainLayout);
        tvMessage = findViewById(R.id.tvMessage);
        btnChangeColor = findViewById(R.id.btnChangeColor);

        btnChangeColor.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Change background color randomly
                mainLayout.setBackgroundColor(Color.parseColor("#FFBB86FC"));
                // Change text color
                tvMessage.setTextColor(Color.parseColor("#FF6200EE"));
            }
        });
    }
}

MainActivity.java
package com.kasinadhuni.passportapp;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

public class MainActivity extends AppCompatActivity {

    EditText etName, etAddress, etNationality, etMobile;
    Spinner spinnerGender;
    Button btnSubmit, btnViewAll;
    SQLiteDatabase db;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.etName);
        etAddress = findViewById(R.id.etAddress);
        etNationality = findViewById(R.id.etNationality);
        etMobile = findViewById(R.id.etMobile);
        spinnerGender = findViewById(R.id.spinnerGender);
        btnSubmit = findViewById(R.id.btnSubmit);
        btnViewAll = findViewById(R.id.btnViewAll);

        db = openOrCreateDatabase("PassportDB", Context.MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS passport(" +
                "name VARCHAR(30), gender VARCHAR(10), address VARCHAR(100), " +
                "nationality VARCHAR(20), mobile INT);");

        btnSubmit.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                submitApplication();
            }
        });

        btnViewAll.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                viewAllApplications();
            }
        });
    }

    private void submitApplication() {
        String name = etName.getText().toString().trim();
        String gender = spinnerGender.getSelectedItem().toString();
        String address = etAddress.getText().toString().trim();
        String nationality = etNationality.getText().toString().trim();
        String mobile = etMobile.getText().toString().trim();

        if(name.isEmpty() || address.isEmpty() || nationality.isEmpty() || mobile.isEmpty()) {
            showMessage("Error", "Please fill all fields");
            return;
        }

        db.execSQL("INSERT INTO passport VALUES('" + name + "','" + gender + "','" +
                address + "','" + nationality + "'," + mobile + ");");

        showMessage("Success", "Passport Application Submitted");
        clearFields();
    }

    private void viewAllApplications() {
        Cursor c = db.rawQuery("SELECT * FROM passport", null);
        if(c.getCount() == 0) {
            showMessage("Error", "No records found");
            return;
        }

        StringBuilder buffer = new StringBuilder();
        while(c.moveToNext()) {
            buffer.append("Name: ").append(c.getString(0)).append("\n");
            buffer.append("Gender: ").append(c.getString(1)).append("\n");
            buffer.append("Address: ").append(c.getString(2)).append("\n");
            buffer.append("Nationality: ").append(c.getString(3)).append("\n");
            buffer.append("Mobile: ").append(c.getInt(4)).append("\n\n");
        }
        showMessage("Passport Applications", buffer.toString());
    }

    private void showMessage(String title, String message) {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setCancelable(true);
        builder.setTitle(title);
        builder.setMessage(message);
        builder.show();
    }

    private void clearFields() {
        etName.setText("");
        etAddress.setText("");
        etNationality.setText("");
        etMobile.setText("");
        spinnerGender.setSelection(0);
    }
}
