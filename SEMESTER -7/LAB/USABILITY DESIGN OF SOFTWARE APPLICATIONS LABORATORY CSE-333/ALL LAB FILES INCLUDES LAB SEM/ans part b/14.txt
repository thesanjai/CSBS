package com.kasinadhuni.set14parta;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.LinearLayout;
import android.widget.ToggleButton;

public class MainActivity extends AppCompatActivity {

    LinearLayout layoutMain;
    ToggleButton toggleButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        layoutMain = findViewById(R.id.layoutMain);
        toggleButton = findViewById(R.id.toggleButton);

        toggleButton.setOnCheckedChangeListener((buttonView, isChecked) -> {
            if(isChecked) {
                layoutMain.setBackgroundColor(getResources().getColor(android.R.color.holo_blue_light));
            } else {
                layoutMain.setBackgroundColor(getResources().getColor(android.R.color.holo_green_light));
            }
        });
    }
}

MainActivity.java
package com.kasinadhuni.set14partb;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {

    EditText etStudentName, etRollNumber, etDepartment, etContact, etAddress;
    Button btnSubmit, btnViewAdmissions;
    SQLiteDatabase db;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etStudentName = findViewById(R.id.etStudentName);
        etRollNumber = findViewById(R.id.etRollNumber);
        etDepartment = findViewById(R.id.etDepartment);
        etContact = findViewById(R.id.etContact);
        etAddress = findViewById(R.id.etAddress);
        btnSubmit = findViewById(R.id.btnSubmit);
        btnViewAdmissions = findViewById(R.id.btnViewAdmissions);

        db = openOrCreateDatabase("CollegeDB", Context.MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS admissions(" +
                "studentName VARCHAR(30), rollNumber INT, department VARCHAR(20), contact INT, address VARCHAR(50));");

        btnSubmit.setOnClickListener(v -> submitAdmission());
        btnViewAdmissions.setOnClickListener(v -> viewAdmissions());
    }

    private void submitAdmission() {
        String name = etStudentName.getText().toString().trim();
        String rollStr = etRollNumber.getText().toString().trim();
        String dept = etDepartment.getText().toString().trim();
        String contactStr = etContact.getText().toString().trim();
        String address = etAddress.getText().toString().trim();

        if(name.isEmpty() || rollStr.isEmpty() || dept.isEmpty() || contactStr.isEmpty() || address.isEmpty()) {
            showMessage("Error", "Please fill all fields");
            return;
        }

        int roll = Integer.parseInt(rollStr);
        int contact = Integer.parseInt(contactStr);

        db.execSQL("INSERT INTO admissions VALUES('" + name + "'," + roll + ",'" + dept + "'," + contact + ",'" + address + "');");
        showMessage("Success", "Admission submitted successfully!");
        clearFields();
    }

    private void viewAdmissions() {
        Cursor c = db.rawQuery("SELECT * FROM admissions", null);
        if(c.getCount() == 0) {
            showMessage("Info", "No admissions found");
            return;
        }

        StringBuilder sb = new StringBuilder();
        while(c.moveToNext()) {
            sb.append("Name: ").append(c.getString(0)).append("\n")
              .append("Roll Number: ").append(c.getInt(1)).append("\n")
              .append("Department: ").append(c.getString(2)).append("\n")
              .append("Contact: ").append(c.getInt(3)).append("\n")
              .append("Address: ").append(c.getString(4)).append("\n\n");
        }

        showMessage("All Admissions", sb.toString());
    }

    private void showMessage(String title, String message) {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setCancelable(true);
        builder.setTitle(title);
        builder.setMessage(message);
        builder.show();
    }

    private void clearFields() {
        etStudentName.setText("");
        etRollNumber.setText("");
        etDepartment.setText("");
        etContact.setText("");
        etAddress.setText("");
    }
}

package com.kasinadhuni.set15partb;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {

    EditText etVehicleNumber, etOwnerName, etModel, etEngineNumber, etMobileNumber;
    Button btnSubmit, btnViewAll;
    SQLiteDatabase db;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etVehicleNumber = findViewById(R.id.etVehicleNumber);
        etOwnerName = findViewById(R.id.etOwnerName);
        etModel = findViewById(R.id.etModel);
        etEngineNumber = findViewById(R.id.etEngineNumber);
        etMobileNumber = findViewById(R.id.etMobileNumber);
        btnSubmit = findViewById(R.id.btnSubmit);
        btnViewAll = findViewById(R.id.btnViewAll);

        db = openOrCreateDatabase("VehicleDB", Context.MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS vehicles(" +
                "vehicleNumber VARCHAR(20), ownerName VARCHAR(30), model VARCHAR(20), engineNumber INT, mobileNumber INT);");

        btnSubmit.setOnClickListener(v -> registerVehicle());
        btnViewAll.setOnClickListener(v -> viewVehicles());
    }

    private void registerVehicle() {
        String vehicleNumber = etVehicleNumber.getText().toString().trim();
        String ownerName = etOwnerName.getText().toString().trim();
        String model = etModel.getText().toString().trim();
        String engineStr = etEngineNumber.getText().toString().trim();
        String mobileStr = etMobileNumber.getText().toString().trim();

        if(vehicleNumber.isEmpty() || ownerName.isEmpty() || model.isEmpty() || engineStr.isEmpty() || mobileStr.isEmpty()) {
            showMessage("Error", "Please fill all fields");
            return;
        }

        int engineNumber = Integer.parseInt(engineStr);
        int mobileNumber = Integer.parseInt(mobileStr);

        db.execSQL("INSERT INTO vehicles VALUES('" + vehicleNumber + "','" + ownerName + "','" + model + "'," + engineNumber + "," + mobileNumber + ");");
        showMessage("Success", "Vehicle Registered Successfully");
        clearFields();
    }

    private void viewVehicles() {
        Cursor c = db.rawQuery("SELECT * FROM vehicles", null);
        if(c.getCount() == 0) {
            showMessage("Info", "No vehicles registered");
            return;
        }

        StringBuilder sb = new StringBuilder();
        while(c.moveToNext()) {
            sb.append("Vehicle Number: ").append(c.getString(0)).append("\n")
              .append("Owner Name: ").append(c.getString(1)).append("\n")
              .append("Model: ").append(c.getString(2)).append("\n")
              .append("Engine Number: ").append(c.getInt(3)).append("\n")
              .append("Mobile Number: ").append(c.getInt(4)).append("\n\n");
        }

        showMessage("Registered Vehicles", sb.toString());
    }

    private void showMessage(String title, String message) {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setCancelable(true);
        builder.setTitle(title);
        builder.setMessage(message);
        builder.show();
    }

    private void clearFields() {
        etVehicleNumber.setText("");
        etOwnerName.setText("");
        etModel.setText("");
        etEngineNumber.setText("");
        etMobileNumber.setText("");
    }
}
