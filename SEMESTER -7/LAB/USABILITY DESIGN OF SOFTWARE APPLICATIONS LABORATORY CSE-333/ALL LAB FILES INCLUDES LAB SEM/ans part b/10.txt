MainActivity.java
package com.kasinadhuni.set10parta;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    TextView tvCounter;
    Button btnIncrease, btnDecrease;
    int counter = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        tvCounter = findViewById(R.id.tvCounter);
        btnIncrease = findViewById(R.id.btnIncrease);
        btnDecrease = findViewById(R.id.btnDecrease);

        btnIncrease.setOnClickListener(v -> {
            counter++;
            tvCounter.setText("Counter: " + counter);
        });

        btnDecrease.setOnClickListener(v -> {
            counter--;
            tvCounter.setText("Counter: " + counter);
        });
    }
}
✅ Logic:
counter variable stores current value.
Buttons update the counter and display it dynamically in the TextView.

MainActivity.java
package com.kasinadhuni.set10partb;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

import java.util.ArrayList;

public class MainActivity extends AppCompatActivity {

    EditText etPatientID, etName, etAge, etGender, etPhone, etAddress;
    Button btnAddPatient, btnViewPatients;
    ListView listViewPatients;
    SQLiteDatabase db;
    ArrayAdapter<String> adapter;
    ArrayList<String> patientList;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etPatientID = findViewById(R.id.etPatientID);
        etName = findViewById(R.id.etName);
        etAge = findViewById(R.id.etAge);
        etGender = findViewById(R.id.etGender);
        etPhone = findViewById(R.id.etPhone);
        etAddress = findViewById(R.id.etAddress);
        btnAddPatient = findViewById(R.id.btnAddPatient);
        btnViewPatients = findViewById(R.id.btnViewPatients);
        listViewPatients = findViewById(R.id.listViewPatients);

        patientList = new ArrayList<>();
        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, patientList);
        listViewPatients.setAdapter(adapter);

        db = openOrCreateDatabase("HospitalDB", Context.MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS patients(" +
                "patientID INT, name VARCHAR(30), age INT, gender VARCHAR(10), phone INT, address VARCHAR(50));");

        btnAddPatient.setOnClickListener(v -> addPatient());
        btnViewPatients.setOnClickListener(v -> viewPatients());
    }

    private void addPatient() {
        String idStr = etPatientID.getText().toString().trim();
        String name = etName.getText().toString().trim();
        String ageStr = etAge.getText().toString().trim();
        String gender = etGender.getText().toString().trim();
        String phoneStr = etPhone.getText().toString().trim();
        String address = etAddress.getText().toString().trim();

        if(idStr.isEmpty() || name.isEmpty() || ageStr.isEmpty() || gender.isEmpty() || phoneStr.isEmpty() || address.isEmpty()) {
            showMessage("Error", "Please fill all fields");
            return;
        }

        int id = Integer.parseInt(idStr);
        int age = Integer.parseInt(ageStr);
        int phone = Integer.parseInt(phoneStr);

        db.execSQL("INSERT INTO patients VALUES(" + id + ",'" + name + "'," + age + ",'" + gender + "'," + phone + ",'" + address + "');");
        showMessage("Success", "Patient added successfully");
        clearFields();
    }

    private void viewPatients() {
        patientList.clear();
        Cursor c = db.rawQuery("SELECT * FROM patients", null);
        if(c.getCount() == 0) {
            showMessage("Error", "No patients found");
            adapter.notifyDataSetChanged();
            return;
        }

        while(c.moveToNext()) {
            String patient = "ID: " + c.getInt(0) +
                    "\nName: " + c.getString(1) +
                    "\nAge: " + c.getInt(2) +
                    "\nGender: " + c.getString(3) +
                    "\nPhone: " + c.getInt(4) +
                    "\nAddress: " + c.getString(5);
            patientList.add(patient);
        }
        adapter.notifyDataSetChanged();
    }

    private void showMessage(String title, String message) {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setCancelable(true);
        builder.setTitle(title);
        builder.setMessage(message);
        builder.show();
    }

    private void clearFields() {
        etPatientID.setText("");
        etName.setText("");
        etAge.setText("");
        etGender.setText("");
        etPhone.setText("");
        etAddress.setText("");
    }
}
