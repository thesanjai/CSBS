package com.kasinadhuni.taxform;

import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

public class MainActivity extends AppCompatActivity implements View.OnClickListener {

    EditText etName, etPAN, etSalary, etCapital, etOther;
    Spinner spinnerFY;
    Button btnAdd, btnDelete, btnModify, btnView, btnViewAll;
    SQLiteDatabase db;

    String[] years = {"2022-23", "2021-22", "2020-21"};

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.etName);
        etPAN = findViewById(R.id.etPAN);
        etSalary = findViewById(R.id.etSalary);
        etCapital = findViewById(R.id.etCapital);
        etOther = findViewById(R.id.etOther);
        spinnerFY = findViewById(R.id.spinnerFY);

        ArrayAdapter<String> adapter = new ArrayAdapter<>(this, android.R.layout.simple_spinner_item, years);
        adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        spinnerFY.setAdapter(adapter);

        btnAdd = findViewById(R.id.btnAdd);
        btnDelete = findViewById(R.id.btnDelete);
        btnModify = findViewById(R.id.btnModify);
        btnView = findViewById(R.id.btnView);
        btnViewAll = findViewById(R.id.btnViewAll);

        btnAdd.setOnClickListener(this);
        btnDelete.setOnClickListener(this);
        btnModify.setOnClickListener(this);
        btnView.setOnClickListener(this);
        btnViewAll.setOnClickListener(this);

        db = openOrCreateDatabase("TaxDB", Context.MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS itreturns(name VARCHAR, pan VARCHAR PRIMARY KEY, salary VARCHAR, capital VARCHAR, other VARCHAR, year VARCHAR);");
    }

    @Override
    public void onClick(View v) {
        if (v == btnAdd) {
            if (!validateInputs()) return;
            db.execSQL("INSERT INTO itreturns VALUES('" + etName.getText() + "','" + etPAN.getText() + "','" + etSalary.getText() + "','" + etCapital.getText() + "','" + etOther.getText() + "','" + spinnerFY.getSelectedItem().toString() + "');");
            showMessage("Success", "Record Added");
            clearText();
        }

        if (v == btnDelete) {
            if (etPAN.getText().toString().trim().isEmpty()) {
                showMessage("Error", "Enter PAN to delete");
                return;
            }
            Cursor c = db.rawQuery("SELECT * FROM itreturns WHERE pan='" + etPAN.getText() + "'", null);
            if (c.moveToFirst()) {
                db.execSQL("DELETE FROM itreturns WHERE pan='" + etPAN.getText() + "'");
                showMessage("Success", "Record Deleted");
            } else {
                showMessage("Error", "Invalid PAN");
            }
            clearText();
        }

        if (v == btnModify) {
            if (!validateInputs()) return;
            Cursor c = db.rawQuery("SELECT * FROM itreturns WHERE pan='" + etPAN.getText() + "'", null);
            if (c.moveToFirst()) {
                db.execSQL("UPDATE itreturns SET name='" + etName.getText() + "', salary='" + etSalary.getText() + "', capital='" + etCapital.getText() + "', other='" + etOther.getText() + "', year='" + spinnerFY.getSelectedItem().toString() + "' WHERE pan='" + etPAN.getText() + "'");
                showMessage("Success", "Record Modified");
            } else {
                showMessage("Error", "Invalid PAN");
            }
            clearText();
        }

        if (v == btnView) {
            if (etPAN.getText().toString().trim().isEmpty()) {
                showMessage("Error", "Enter PAN to view");
                return;
            }
            Cursor c = db.rawQuery("SELECT * FROM itreturns WHERE pan='" + etPAN.getText() + "'", null);
            if (c.moveToFirst()) {
                etName.setText(c.getString(0));
                etSalary.setText(c.getString(2));
                etCapital.setText(c.getString(3));
                etOther.setText(c.getString(4));
                // year shown via spinner automatically
            } else {
                showMessage("Error", "Invalid PAN");
                clearText();
            }
        }

        if (v == btnViewAll) {
            Cursor c = db.rawQuery("SELECT * FROM itreturns", null);
            if (c.getCount() == 0) {
                showMessage("Error", "No records found");
                return;
            }
            StringBuilder buffer = new StringBuilder();
            while (c.moveToNext()) {
                buffer.append("Name: ").append(c.getString(0)).append("\n");
                buffer.append("PAN: ").append(c.getString(1)).append("\n");
                buffer.append("Salary: ").append(c.getString(2)).append("\n");
                buffer.append("Capital: ").append(c.getString(3)).append("\n");
                buffer.append("Other: ").append(c.getString(4)).append("\n");
                buffer.append("Year: ").append(c.getString(5)).append("\n\n");
            }
            showMessage("IT Returns", buffer.toString());
        }
    }

    private boolean validateInputs() {
        if (etName.getText().toString().trim().isEmpty() ||
                etPAN.getText().toString().trim().isEmpty() ||
                etSalary.getText().toString().trim().isEmpty()) {
            showMessage("Error", "Name, PAN and Salary are mandatory");
            return false;
        }
        if (etPAN.getText().toString().trim().length() != 10) {
            showMessage("Error", "PAN must be 10 characters");
            return false;
        }
        return true;
    }

    private void showMessage(String title, String msg) {
        AlertDialog.Builder b = new AlertDialog.Builder(this);
        b.setCancelable(true);
        b.setTitle(title);
        b.setMessage(msg);
        b.show();
    }

    private void clearText() {
        etName.setText("");
        etPAN.setText("");
        etSalary.setText("");
        etCapital.setText("");
        etOther.setText("");
        etName.requestFocus();
    }
}

package com.kasinadhuni.chatapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.Menu;
import android.view.MenuInflater;
import android.view.MenuItem;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
    }

    // Inflate the menu
    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        MenuInflater inflater = getMenuInflater();
        inflater.inflate(R.menu.chat_menu, menu); // linking to chat_menu.xml
        return true;
    }

    // Handle menu clicks
    @Override
    public boolean onOptionsItemSelected(MenuItem item) {
        switch (item.getItemId()) {
            case R.id.action_new_group:
                Toast.makeText(this, "New Group clicked", Toast.LENGTH_SHORT).show();
                // ðŸ‘‰ Here you can open New Group Activity
                return true;

            case R.id.action_new_message:
                Toast.makeText(this, "New Message clicked", Toast.LENGTH_SHORT).show();
                // ðŸ‘‰ Here you can open New Message Activity
                return true;

            case R.id.action_starred:
                Toast.makeText(this, "Starred Messages clicked", Toast.LENGTH_SHORT).show();
                // ðŸ‘‰ Here you can open Starred Messages screen
                return true;

            default:
                return super.onOptionsItemSelected(item);
        }
    }
}
